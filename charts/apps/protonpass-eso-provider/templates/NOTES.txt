Use it in your ExternalSecret resources:

  apiVersion: external-secrets.io/v1
  kind: ExternalSecret
  metadata:
    name: my-app-secrets
  spec:
    refreshInterval: "5m"
    secretStoreRef:
      name: {{ .Values.externalSecretsOperator.secretStore.name }}
      kind: {{ .Values.externalSecretsOperator.namespaced | ternary "SecretStore" "ClusterSecretStore" }}
    data:
      - secretKey: DB_PASSWORD
        remoteRef:
          key: "my-database-item"       # Proton Pass item title or ID
          property: "password"           # Field name (password, username, etc.)
      - secretKey: API_KEY
        remoteRef:
          key: "my-api-keys"
          property: "api_key"            # Custom field name

{{- end }}
